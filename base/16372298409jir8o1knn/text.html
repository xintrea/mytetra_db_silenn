void RCC_init(void)

{

RCC->CR |= (RCC_CR_HSEON);

while(!(RCC->CR & RCC_CR_HSERDY)) {}

FLASH->ACR |= (1<<FLASH_ACR_LATENCY);

FLASH->ACR |= (FLASH_ACR_PRFTBE);

RCC->CR &= ~(RCC_CR_PLLON); //disable PLL before any changes

RCC->CFGR |= (RCC_CFGR_PLLSRC);

RCC->CFGR |= (RCC_CFGR_PLLMUL6);

RCC->CFGR &= ~(RCC_CFGR_PPRE); //PCLK = 1 (for APB)

RCC->CFGR &= ~(RCC_CFGR_HPRE); //HCLK = 1 (for AHB)

//RCC->CFGR2 |= (RCC_CFGR2_PREDIV); //PREDIV = 1

RCC->CR |= (RCC_CR_PLLON);

while(!(RCC->CR & RCC_CR_PLLRDY)) {}

RCC->CFGR |= (RCC_CFGR_SW_PLL); //setting PLL as source clock

while(!(RCC->CFGR & RCC_CFGR_SWS_PLL)) {}

}


#define TIM17_10_MSEC (10-1)

#define TIM17_50_MSEC (50-1)

#define TIM17_100_MSEC (100-1)


void TIM17_init(void)

{

RCC->APB2ENR |= RCC_APB2ENR_TIM17EN;

TIM17->PSC = 47999; //48MHz. 1 interrupt per 1 ms if ARR == 0

TIM17->ARR = TIM17_10_MSEC; //10 ms

TIM17->DIER |= TIM_DIER_UIE; //update interrupt enable

NVIC_EnableIRQ(TIM17_IRQn);


TIM17->CR1 |= TIM_CR1_CEN; //counter enable

}
