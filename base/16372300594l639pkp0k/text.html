// *** LCD based on HD44780

// 4-bit mode

//

// PB1 - RS

// PA4 - E

//

// PA6 - Data pin 7

// PA5 - Data pin 6

// PA1 - Data pin 5

// PA0 - Data pin 4

//----------------------------------

/*******************************

Changeable Params (Pins and Ports)

******************************/

#define LCD_RS_PORT GPIOB

#define LCD_RS 1

#define LCD_E_PORT GPIOA

#define LCD_E 4

#define LCD_RS_UP GPIO_BSRR_BS_1

#define LCD_RS_DWN GPIO_BSRR_BR_1

#define LCD_E_UP GPIO_BSRR_BS_4

#define LCD_E_DWN GPIO_BSRR_BR_4


#define LCD_DATPORT GPIOA

#define LCD_DATMASK 0x00000063

#define LCD_D7 6

#define LCD_D6 5

#define LCD_D5 1

#define LCD_D4 0

#define LCD_D7_UP GPIO_BSRR_BS_6

#define LCD_D7_DWN GPIO_BSRR_BR_6

#define LCD_D6_UP GPIO_BSRR_BS_5

#define LCD_D6_DWN GPIO_BSRR_BR_5

#define LCD_D5_UP GPIO_BSRR_BS_1

#define LCD_D5_DWN GPIO_BSRR_BR_1

#define LCD_D4_UP GPIO_BSRR_BS_0

#define LCD_D4_DWN GPIO_BSRR_BR_0

/*******************************

END of Changeable Params (Pins and Ports)

******************************/


#define LCD_RS_SET LCD_RS_PORT->BSRR |= LCD_RS_UP

#define LCD_RS_CLR LCD_RS_PORT->BSRR |= LCD_RS_DWN

#define LCD_E_SET LCD_E_PORT->BSRR |= LCD_E_UP

#define LCD_E_CLR LCD_E_PORT->BSRR |= LCD_E_DWN

#define LCD_E_STATE (LCD_E_PORT->IDR & LCD_E)


#define LCD_D7_SET LCD_DATPORT->BSRR |= LCD_D7_UP

#define LCD_D7_CLR LCD_DATPORT->BSRR |= LCD_D7_DWN

#define LCD_D6_SET LCD_DATPORT->BSRR |= LCD_D6_UP

#define LCD_D6_CLR LCD_DATPORT->BSRR |= LCD_D6_DWN

#define LCD_D5_SET LCD_DATPORT->BSRR |= LCD_D5_UP

#define LCD_D5_CLR LCD_DATPORT->BSRR |= LCD_D5_DWN

#define LCD_D4_SET LCD_DATPORT->BSRR |= LCD_D4_UP

#define LCD_D4_CLR LCD_DATPORT->BSRR |= LCD_D4_DWN

#define LCD_DAT_CLR LCD_DATPORT->ODR &= ~LCD_DATMASK


//positions

#define LCD_CLEAR 0x01

#define LCD_HOME 0x02

#define LCD_MONTH_POS 0x83

#define LCD_YEAR_POS 0x86

#define LCD_SECOND_LINE 0xC0

#define LCD_MIN_POS 0xC3

#define LCD_SEC_POS 0xC6


//commands

#define LCD_ON 0x0C

#define LCD_OFF 0x08


//symbols

#define LCD_ZERO 0x30

#define LCD_DOTS 0x0A

#define LCD_POINT 0x2E

static __forceinline void HD44780_GPIO_init(void)

{

RCC->AHBENR |= RCC_AHBENR_GPIOAEN;

RCC->AHBENR |= RCC_AHBENR_GPIOBEN;

GPIOA->MODER &= ~GPIO_MODER_MODER0;

GPIOA->MODER |= GPIO_MODER_MODER0_0;

GPIOA->MODER &= ~GPIO_MODER_MODER1;

GPIOA->MODER |= GPIO_MODER_MODER1_0;

GPIOA->MODER &= ~GPIO_MODER_MODER4;

GPIOA->MODER |= GPIO_MODER_MODER4_0;

GPIOA->MODER &= ~GPIO_MODER_MODER5;

GPIOA->MODER |= GPIO_MODER_MODER5_0;

GPIOA->MODER &= ~GPIO_MODER_MODER6;

GPIOA->MODER |= GPIO_MODER_MODER6_0;

GPIOA->OSPEEDR |= GPIO_OSPEEDER_OSPEEDR0;

GPIOA->OSPEEDR |= GPIO_OSPEEDER_OSPEEDR1;

GPIOA->OSPEEDR |= GPIO_OSPEEDER_OSPEEDR4;

GPIOA->OSPEEDR |= GPIO_OSPEEDER_OSPEEDR5;

GPIOA->OSPEEDR |= GPIO_OSPEEDER_OSPEEDR6;


GPIOB->MODER &= ~GPIO_MODER_MODER1;

GPIOB->MODER |= GPIO_MODER_MODER1_0;

GPIOB->OSPEEDR |= GPIO_OSPEEDER_OSPEEDR1;

}


static __forceinline void HD44780_init(void)

{

LCD_DAT_CLR;

LCD_E_SET;

wait_10ms(4);


lcdSendNibble(0x30); //8 bit mode ON

wait_10ms(1);

lcdSendNibble(0x30); //8 bit mode ON

wait_10ms(1);

lcdSendNibble(0x30); //8 bit mode ON

wait_10ms(1);

lcdSendNibble(0x20); //4 bit mode ON

wait_10ms(1);


lcdSendCmd(0x28); //4 bit mode ON, 2 lines

lcdSendCmd(0x08); //display OFF

lcdSendCmd(0x01); //clear display

lcdSendCmd(0x06); //incr cursor position


lcdSendCmd(0x0C); //display ON without cursor

// lcdSendCmnd(0x0F); //display ON with cursor


wait_10ms(2);

lcdSendCmd(0x01); //clear display

lcdSendCmd(0x01);

lcdSendCmd(0x02);

}

void lcdSendNibble(const u8 cmnd)

{

LCD_DAT_CLR;

LCD_E_SET;

if(cmnd & (1<<7)) LCD_D7_SET;

if(cmnd & (1<<6)) LCD_D6_SET;

if(cmnd & (1<<5)) LCD_D5_SET;

if(cmnd & (1<<4)) LCD_D4_SET;

wait_10ms(1); // <---------------------- VERY IMPORTANT DELAY

LCD_E_CLR;

}

/*****************************************************************************

* @name :void initLCD(void)

* @date :05-10-2021

* @function :HD44780 initialization

* @parameters :none

* @retvalue :none

******************************************************************************/

void initLCD(void)

{

HD44780_GPIO_init();

HD44780_init();

}


/*****************************************************************************

* @name :void lcdSendCmd(const u8 cmnd)

* @date :05-10-2021

* @function :sending command (without RS setting) in 4-bits mode

* @parameters :8-bits command

* @retvalue :none

******************************************************************************/

void lcdSendCmd(const u8 cmnd)

{

lcdSendNibble(cmnd);

lcdSendNibble(cmnd<<4);

LCD_DAT_CLR;

}


/*****************************************************************************

* @name :void lcdSendChar(const u8 data)

* @date :05-10-2021

* @function :sending one char symbol for display

* @parameters :char symbol

* @retvalue :none

******************************************************************************/

void lcdSendChar(const u8 data)

{

LCD_RS_SET;

lcdSendCmd(data);

LCD_RS_CLR;

}


/*****************************************************************************

* @name :void lcdSendString(const char* str)

* @date :05-10-2021

* @function :sending a string for display

* @parameters :pointer to string

* @retvalue :none

******************************************************************************/

void lcdSendString(const char* str)

{

uint8_t i = 0;

while(str[i])

{

lcdSendChar(str[i++]);

}

}


/*****************************************************************************

* @name :void lcdSendDig(const u8 data)

* @date :05-10-2021

* @function :sending one digit symbol for display

* @parameters :digit symbol

* @retvalue :none

******************************************************************************/

void lcdSendDig(const u8 data)

{

LCD_RS_SET;

lcdSendCmd(data + LCD_ZERO); //+0x30

LCD_RS_CLR;

}


/*****************************************************************************

* @name :void lcdSendNum(i32 data)

* @date :05-10-2021

* @function :sending for display a positive or negative number

with 6 digits as maximum

* @parameters :number for display

* @retvalue :none

******************************************************************************/

void lcdSendNum(i32 data)

{

static const u32 dec = 10;

u8 number[4];

i8 digit = -1;

if(data < 0)

{

lcdSendChar('-');

data = -data;

}

while(data)

{

if(data >= dec)

{

u8 val = data % dec;

data /= dec;

number[++digit] = val;

}

else

{

break;

}

}

lcdSendDig(data);

while(digit >= 0)

{

lcdSendDig(number[digit]);

--digit;

}
