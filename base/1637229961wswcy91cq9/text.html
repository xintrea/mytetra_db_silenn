/*****************************************************************************

* @brief I2C1 initialization

* @param none

* @return none

******************************************************************************/

void I2C1_init(void)

{

RCC->AHBENR |= RCC_AHBENR_GPIOAEN;

GPIOA->MODER &= ~(GPIO_MODER_MODER9);

GPIOA->MODER |= (GPIO_MODER_MODER9_1); //alternate mode

GPIOA->MODER &= ~(GPIO_MODER_MODER10);

GPIOA->MODER |= (GPIO_MODER_MODER10_1); //alternate mode

GPIOA->OTYPER |= GPIO_OTYPER_OT_9 | GPIO_OTYPER_OT_10; //open-drain

GPIOA->OSPEEDR |= GPIO_OSPEEDER_OSPEEDR9; //fast speed

GPIOA->OSPEEDR |= GPIO_OSPEEDER_OSPEEDR10; //fast speed

GPIOA->AFR[1] |= (4 << 4); //enable AF for PA9

GPIOA->AFR[1] |= (4 << 8); //enable AF for PA10

RCC->APB1ENR |= (RCC_APB1ENR_I2C1EN); //I2C1 clock enable

while ((RCC->APB1ENR & RCC_APB1ENR_I2C1EN) == 0);

RCC->CFGR3 |= (RCC_CFGR3_I2C1SW); //clock source - sysclock

I2C1->CR1 &= ~(I2C_CR1_PE); //disable I2C

RCC->APB1RSTR |= RCC_APB1RSTR_I2C1RST; //reset I2C1

RCC->APB1RSTR &= ~RCC_APB1RSTR_I2C1RST;

//I2C1->CR1 |= (I2C_CR1_TXIE); //transmit interrupt enabled

//I2C1->CR1 |= (I2C_CR1_RXIE); //receive interrupt enabled

I2C1->TIMINGR = 0x00B01A4B;

//I2C1->TIMINGR = 0x00601B53;

//I2C1->TIMINGR = 0xB0420F13;


I2C1->CR2 |= (I2C_CR2_AUTOEND);

}


/*****************************************************************************

* @brief I2C1 1 byte master transfer to slave

* @param addr - destination slave address

* @param byte - data byte

* @return none

******************************************************************************/

void i2cMasterWrByte(const u8 addr, const u8 byte)

{

if(I2C1->ISR & I2C_ISR_BUSY)

{

return; //I2C is busy

}

I2C1->CR1 |= (I2C_CR1_PE); //enable I2C


I2C1->CR2 |= ((addr << 1) | I2C_RD_DIRECTION) | //sets an address and direction bit for slav

(1 << 16) | //sets NBYTES

I2C_CR2_START; //START condition


if((I2C1->ISR & I2C_ISR_TXE) == (I2C_ISR_TXE)) /* Check Tx empty */

{

I2C1->TXDR = byte; /* Byte to send */

}

while(!(I2C1->ISR & I2C_ISR_STOPF)) {} //waiting for stop condition

I2C1->CR2 &= ~(I2C_CR2_SADD | I2C_CR2_HEAD10R | I2C_CR2_NBYTES | I2C_CR2_RELOAD | I2C_CR2_RD_WRN); //clear CR2:

I2C1->CR1 &= ~(I2C_CR1_PE); //disable I2C

}


/*****************************************************************************

* @brief I2C1 packet master transfer to slave

* @param addr - destination slave address

* @param data - pointer of data packet to send

* @param bytes - amount of bytes

* @return none

******************************************************************************/

void i2cMasterWrite(const u8 addr, const u8* data, const u8 bytes)

{

if(I2C1->ISR & I2C_ISR_BUSY)

{

return; //I2C is busy

}

I2C1->CR1 |= (I2C_CR1_PE); //enable I2C


I2C1->CR2 |= ((addr << 1) | I2C_RD_DIRECTION) | //sets an address and direction bit for slav

(bytes << 16) | //sets NBYTES

I2C_CR2_START; //START condition


for(u8 i=0; i<bytes; ++i)

{

while(!((I2C1->ISR & I2C_ISR_TXE) == (I2C_ISR_TXE))) {} //TXE==1 if TXDR is empty

I2C1->TXDR = data[i]; //puts data in TX register

}

while(!(I2C1->ISR & I2C_ISR_STOPF)) {} //waiting for stop condition

I2C1->CR2 &= ~(I2C_CR2_SADD | I2C_CR2_HEAD10R | I2C_CR2_NBYTES | I2C_CR2_RELOAD | I2C_CR2_RD_WRN); //clear CR2

I2C1->CR1 &= ~(I2C_CR1_PE); //disable I2C

}


/*****************************************************************************

* @brief I2C1 master receiving data from slave

* @param addr - destination slave address

* @param data - pointer for data storing

* @param bytes - amount of bytes

* @return none

******************************************************************************/

void i2cMasterRead(const u8 addr, u8* data, const u8 bytes)

{

if(I2C1->ISR & I2C_ISR_BUSY)

{

return; //I2C is busy

}

I2C1->CR1 |= (I2C_CR1_PE); //enable I2C


I2C1->CR2 |= ((addr << 1) | I2C_RD_DIRECTION) | //sets an address and direction bit for slav

(bytes << 16) | //sets NBYTES

I2C_CR2_RD_WRN | //requests a read transfer

I2C_CR2_START; //START condition


for(u8 i=0; i<bytes; ++i)

{

while(!((I2C1->ISR & I2C_ISR_RXNE) == I2C_ISR_RXNE)) //RXNE==1 if RXDR is NOT empty

{}

data[i] = I2C1->RXDR; //puts data in TX register

}

while(!(I2C1->ISR & I2C_ISR_STOPF)) {} //waiting for stop condition

I2C1->CR2 &= ~(I2C_CR2_SADD | I2C_CR2_HEAD10R | I2C_CR2_NBYTES | I2C_CR2_RELOAD | I2C_CR2_RD_WRN); //clear CR2

I2C1->CR1 &= ~(I2C_CR1_PE); //disable I2C

}
